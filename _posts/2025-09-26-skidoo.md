Hey there!

<button id="cool-button">Press this button to learn cool stuff!</button>

<div id="skidoo-content"></div>

<script type="module">
    import { pythonURI, fetchOptions } from "{{ site.baseurl }}/assets/js/api/config.js";

    document.getElementById("cool-button").addEventListener("click", () => {
        fetch(pythonURI + "/getSkidoo", fetchOptions)
            .then(res => res.json())
            .then(data => {
                const container = document.getElementById("skidoo-content");
                container.innerHTML = `<h2>${data.title}</h2><p>${data.body}</p>`;
            });
    });
</script>
